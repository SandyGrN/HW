
CREATE TABLE airbnb.users (
	user_id INT IDENTITY (1, 1) PRIMARY KEY,
	user_name VARCHAR (255) NOT NULL,
	phone_number VARCHAR (25),
	email VARCHAR (255),
    email_verified_at 
	street VARCHAR (255),
	city VARCHAR (255),
	state VARCHAR (10),
	zip_code VARCHAR (5)
);



CREATE TABLE airbnb.reservations (
	reservation_id INT IDENTITY (1, 1) PRIMARY KEY,
	guest_id INT (50) NOT NULL,
	apartments_id INT (50) NOT NULL,
	payment_id INT (50) NOT NULL,
	status VARCHAR,
	start_date DATETIME,
	end_date DATETIME, 
	price INT NOT NULL,
	guests_amount INT NOT NULL,
	total_price INT NOT NULL,
	FOREIGN KEY (guest_id) 
        REFERENCES airbnb.users  (user_id) 
	FOREIGN KEY (apartments_id) 
        REFERENCES airbnb.apartments (apartment_id) 
	FOREIGN KEY (payment_id) 
        REFERENCES sales.paymrnts (payment_id) 
);
